	\begin{exercicio}
		Produza cada uma das tabelas abaixo separadamente. Depois, organize-as conforme abaixo (ie, colocando-as na mesma linha e equidistantes).
		
		\medskip
		
		\noindent\hfill
		\begin{tabular}{cc}		
			1 & 2 \\		
			3 & 4 \\
		\end{tabular}\hfill
		\begin{tabular}{|c|c|}
			1 & 2 \\ 
			3 & 4 \\ 
		\end{tabular}\hfill
		\begin{tabular}{cc}
			\hline
			1 & 2 \\ 
			\hline
			3 & 4 \\
			\hline
		\end{tabular}\hfill
		\begin{tabular}{|c|c|}
			\hline
			1 & 2 \\ 
			\hline
			3 & 4 \\ 
			\hline
		\end{tabular}\hfill\null
		
		\medskip
		
		\footnotesize
		\noindent\textbf{obs.:} use \cs{hfill} entre as tabelas e, após a última, 
		use \cs{hfill}\cs{null} ou \cs{hfill\{\}}. Isto é necessário porque o 
		\LaTeX\ ignora espaços horizontais no final da linha (ou espaços verticais 
		no final da página).
	\end{exercicio}

	\begin{exercicio}
		Utilize os comandos \cs{raisebox} e \cs{includegraphics} para produzir o 
		resultado a seguir (a linha horizontal representa a linha-base):
		
		\medskip
		
		\noindent
		\hrulefill\raisebox{+0.0\height}{\includegraphics[width=1cm]{Tei-Gi}}%
		\hrulefill\raisebox{-0.5\height}{\includegraphics[width=1cm]{Tei-Gi}}%
		\hrulefill\raisebox{-1.0\height}{\includegraphics[width=1cm]{Tei-Gi}}%
		\hrulefill\null
		
		\medskip
		
		\footnotesize
		\noindent\textbf{dica:} quando você cria uma caixa LR com os comandos
		\cs{raisebox}, \cs{framebox} ou \cs{makebox}, o \LaTeX\ define
		automaticamente os seguintes comprimentos: \cs{width}, \cs{height}, 
		\cs{depth} e \cs{totalheight}, que representam a largura, altura,
		profundidade e altura-total \emph{natural} desta caixa,	respectivamente.
		Mas atenção: esses comprimentos só valem dentro dos	argumentos (opcionais
		ou obrigatórios) do comando. Por exemplo, eu posso definir	uma caixa LR
		de comprimento igual ao dobro do seu comprimento natural assim: 
		\cs{framebox}\verb|[2|\cs{width}\verb|]{|\textit{\textbf{caixa}}\verb|}|.
	\end{exercicio}

	\begin{exercicio}
		\textbf{Desafio:} construa a célula unificada da tabela abaixo sem
		utilizar o pacote \pkg{multirow}.

		\setlength{\fboxsep}{0.5pt}

		\medskip

		\begin{center}
			\begin{tabular}{|c|c|}		
				\hline
				Célula 11 &\\
				\cline{1-1}
				Célula 21 & \rule{3mm}{0pt}%
					\raisebox{1ex}[\height][0pt]{União}%
					\rule{3mm}{0pt} \\
				\hline
			\end{tabular}
		\end{center}
		
		\medskip
		
		\footnotesize
		\noindent\textbf{dica:} a célula unificada em questão não é, de fato,
	unificada. Acontece que a célula $(\text{linha},\text{coluna}) = (1,2)$ está 
	vazia, enquanto $(2,2)$ contém um \cs{raisebox}. Experimente.

	\end{exercicio}
